\section*{実機デバッグの設定}
\medskip

\begin{itemize}
\item /etc/udev/rules.d/51-android.rules の作成\\
{\tiny \begin{verbatim}SUBSYSTEM=="usb", SYSFS{idVendor}=="0bb4", MODE="0666"\end{verbatim}}
